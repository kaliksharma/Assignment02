package javaapplication1;
public class Movies
{
   String movie_name;
   int year;
   int Rating;
    Movies(int year,int rating,String movie_name)
    {
       this.year=year;
       this.Rating=rating;
       this.movie_name=movie_name;
    }

    public String getMovie_name() 
     {
        return movie_name;
     } 

     public int getYear() 
     {
        return year;
     }

    public int getRating() 
    {
        return Rating;
    }
 
}
